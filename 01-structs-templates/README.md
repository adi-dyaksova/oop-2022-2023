## Шаблонни функции

```c++
void printInt(int a)
{
    std::cout << a;
}

void printDouble(double a)
{
    std::cout << a;
}

void printChar(char a)
{
    std::cout << a;
}

template <typename T>
void print(T a)
{
    std::cout << a;
}


int main()
{
    int n = 1;
    double d = 1.5;
    char c = 'b';
    printInt(n);
    printDouble(d);
    printChar(c);

    print<int>(n);
    print<double>(d);
    print<char>(c);
}
```
